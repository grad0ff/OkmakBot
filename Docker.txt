*Ubuntu 18.04*

- обновить систему:
sudo apt update

- установить curl:
sudo apt install curl

- установить docker:
curl -fsSL https://get.docker.com -o get-docker.sh
sudo sh get-docker.sh

- включить автозапуск docker (кроме Ubuntu):
sudo systemctl enable docker.service
sudo systemctl enable containerd.service

- cоздать Dockerfile
FROM python:3.9-slim
WORKDIR /home/projects/okmakbot
RUN apt-get update
#RUN apt-get upgrade
RUN apt-get install -y python3-pip
RUN pip3 install aiogram
RUN apt-get clean && \
    apt-get -y autoclean && \
    apt-get -y autoremove
CMD ['python3', 'main.py']

- cоздать образ на основе Dockerfile
sudo docker build -t okmakbot_img .

- запуск контейнера
sudo docker run -it --restart=always -e DIRECTORY='/home/projects/okmakbot' -v /home/projects/okmakbot/:/dir okmakbot_img python3 /dir/main.py


# sudo docker run -d --restart=unless-stopped -v /home/projects/okmakbot/:/dir okmakbot_img python3 /dir/main.py
